Set up commands using Anaconda Prompt:
cd /d D:\Anaconda3
conda create -c rdkit -n my-rdkit-env rdkit

activate my-rdkit-env
[cd into script folder]
python test_rdkit.py

activate my-rdkit-env
[cd into script folder]
python test_rdkit.py
